package cn.edu.xmu.oomall.service.dao;


import cn.edu.xmu.javaee.core.model.UserToken;
import cn.edu.xmu.oomall.service.dao.bo.ServiceOrder;
import cn.edu.xmu.oomall.service.mapper.ServiceOrderPoMapper;
import cn.edu.xmu.oomall.service.mapper.po.ServiceOrderPo;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.jdbc.AutoConfigureTestDatabase;
import org.springframework.boot.test.autoconfigure.orm.jpa.DataJpaTest;
import org.springframework.context.annotation.Import;
import org.springframework.test.context.ActiveProfiles;

import java.time.LocalDateTime;

import static org.assertj.core.api.Assertions.assertThat;
import static org.junit.jupiter.api.Assertions.assertAll;

@DataJpaTest  // 仅启动 JPA 相关组件，速度快且自动回滚
@Import({ServiceOrderDao.class,
        ServiceProviderDao.class}) // 注入被测类
@AutoConfigureTestDatabase(replace = AutoConfigureTestDatabase.Replace.NONE)
@DisplayName("服务单Dao层补充测试")
public class ServiceOrderDaoTest {

    @Autowired
    private ServiceOrderDao serviceOrderDao;

    @Autowired
    private ServiceOrderPoMapper serviceOrderPoMapper;

    private UserToken testUser;
    private ServiceOrder testBo;

    @BeforeEach
    void setUp() {
        // 准备用户信息
        testUser = new UserToken();
        testUser.setId(123L);
        testUser.setName("测试人员");

        // 准备初始 BO 数据
        testBo = new ServiceOrder();
        testBo.setShopId(100L);
        testBo.setConsignee("张三");
        testBo.setMobile("13900139000");
        testBo.setAddress("厦门大学");
        testBo.setRegionId(1101L);
    }

    @Test
    @DisplayName("数据库增加服务单")
    void insert_Success() {
        // 1. 执行插入
        ServiceOrder result = serviceOrderDao.insert(testBo, testUser);

        // 2. 验证返回的 BO
        assertAll("验证返回对象",
                () -> assertThat(result.getId()).isNotNull(), // ID 应该已被数据库填充
                () -> assertThat(result.getCreatorId()).isEqualTo(testUser.getId()), // 验证 Creator
                () -> assertThat(result.getGmtCreate()).isBeforeOrEqualTo(LocalDateTime.now()),
                () -> assertThat(result.getConsignee()).isEqualTo("张三")
        );

        // 3. 验证数据库中是否真的存在
        ServiceOrderPo savedPo = serviceOrderPoMapper.findById(result.getId()).orElse(null);
        assertThat(savedPo).isNotNull();
        assertThat(savedPo.getShopId()).isEqualTo(100L);
        assertThat(savedPo.getAddress()).isEqualTo("厦门大学");
    }
}
